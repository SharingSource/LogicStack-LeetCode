    Map<Integer, List<Integer>> map = new HashMap<>();
    public Solution(int[] nums) {
        int n = nums.length;
        for (int i = 0; i < n; i++) {

            map.put(nums[i], list);
        }
    }

        return list.get(random.nextInt(list.size()));
    }
}
